RIGHT JOIN
select 
e.FIRST_NAME,
d.DEPARTMENT_NAME,
d.DEPARTMENT_ID
from 
HR.EMPLOYEES e
RIGHT join HR.DEPARTMENTS d on d.DEPARTMENT_ID = e.DEPARTMENT_ID
where e.DEPARTMENT_ID is null

LEFT JOIN
select 
e.FIRST_NAME,
d.DEPARTMENT_NAME,
d.DEPARTMENT_ID
from 
HR.EMPLOYEES e
LEFT join HR.DEPARTMENTS d on d.DEPARTMENT_ID = e.DEPARTMENT_ID
where e.DEPARTMENT_ID is null
